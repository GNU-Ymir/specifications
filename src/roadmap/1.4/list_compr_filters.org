Filters with the keyword /if/.

#+BEGIN_SRC ymir
fn foo (x: i32)-> i32 {
   x^^2
}
fn filter(x: i32)-> bool {
   (x % 2) == 0
}

let a = copy [x => foo(x) for x in 0 .. 10 if filter(x)];
let b = copy [i for i in 0 .. 10 if i != 6];
